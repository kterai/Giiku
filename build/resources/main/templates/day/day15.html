統合演習シナリオ

演習1: 学生成績管理システム

// 学生情報を管理し、様々な条件での分析を行う
// 1. ジェネリクスを使った型安全なデータ構造
public class Student<T extends Number> {
    private String name;
    private Map<String, T> scores;
    // コンストラクタとメソッド
}

// 2. Stream APIを使った成績分析
students.stream()
    .filter(s -> s.getAverageScore() > 80)
    .collect(Collectors.groupingBy(
        Student::getGrade,
        Collectors.averagingDouble(Student::getAverageScore)
    ));

演習2: オンラインショップ商品管理

// 商品データの処理とレポート生成
// 1. ラムダ式を使った動的フィルタリング
Predicate<Product> priceFilter = p -> p.getPrice() > 1000;
Predicate<Product> categoryFilter = p -> p.getCategory().equals("Electronics");

// 2. 複合処理での Stream API 活用
Map<String, List<Product>> productsByCategory = products.stream()
    .filter(priceFilter.and(categoryFilter))
    .collect(Collectors.groupingBy(Product::getCategory));

演習3: ログ分析システム

// ログデータの解析とレポート作成
// 1. 関数型インターフェースを活用した処理パイプライン
Function<String, LogEntry> parser = line -> LogEntry.parse(line);
Consumer<LogEntry> validator = entry -> entry.validate();

// 2. 大量データの効率的処理
logs.parallelStream()
    .map(parser)
    .peek(validator)
    .filter(entry -> entry.getLevel() == LogLevel.ERROR)
    .collect(Collectors.toMap(
        LogEntry::getTimestamp,
        LogEntry::getMessage,
        (existing, replacement) -> existing
    ));
