ES6+モダンJavaScript機能

アロー関数（Arrow Functions）

アロー関数は従来の function キーワードより簡潔な記法で、thisの扱いが異なる特徴があります。

従来の関数記法

// 従来の関数宣言
function add(a, b) {
    return a + b;
}

// 従来の関数式
const multiply = function(a, b) {
    return a * b;
};

// 配列処理での使用
const numbers = [1, 2, 3, 4, 5];
const doubled = numbers.map(function(num) {
    return num * 2;
});

アロー関数記法

// アロー関数（基本形）
const add = (a, b) => {
    return a + b;
};

// 単一式の場合はreturn省略可能
const multiply = (a, b) => a * b;

// 配列処理での使用（簡潔）
const numbers = [1, 2, 3, 4, 5];
const doubled = numbers.map(num => num * 2);

分割代入（Destructuring Assignment）

分割代入は配列やオブジェクトから値を取り出して、別々の変数に代入する構文です。

配列の分割代入

// 基本的な配列分割代入
const fruits = ['apple', 'banana', 'orange'];
const [first, second, third] = fruits;
console.log(first);  // 'apple'
console.log(second); // 'banana'

// デフォルト値の設定
const [a, b, c = 'default'] = ['x', 'y'];
console.log(c); // 'default'

// 残り要素の取得
const numbers = [1, 2, 3, 4, 5];
const [head, ...tail] = numbers;
console.log(head); // 1
console.log(tail); // [2, 3, 4, 5]

オブジェクトの分割代入

// 基本的なオブジェクト分割代入
const user = {
    name: '田中太郎',
    age: 25,
    email: 'tanaka@example.com'
};

const {name, age, email} = user;
console.log(name); // '田中太郎'

// 別名での代入
const {name: userName, age: userAge} = user;
console.log(userName); // '田中太郎'

// ネストされたオブジェクト
const profile = {
    user: {name: '佐藤', details: {city: '東京'}}
};
const {user: {name: profileName, details: {city}}} = profile;

テンプレートリテラル（Template Literals）

テンプレートリテラルはバッククォート（`）を使用して、文字列内に変数や式を埋め込める機能です。

従来の文字列結合

// 従来の文字列結合
const name = '田中太郎';
const age = 25;
const message = 'こんにちは、' + name + 'さん。' +
                '年齢は' + age + '歳ですね。';

// 複数行文字列
const html = '<div>' +
             '  <h1>タイトル</h1>' +
             '  <p>内容</p>' +
             '</div>';

テンプレートリテラル

// テンプレートリテラル
const name = '田中太郎';
const age = 25;
const message = `こんにちは、${name}さん。
年齢は${age}歳ですね。`;

// 複数行文字列（自然な改行）
const html = `<div>
  <h1>タイトル</h1>
  <p>内容</p>
</div>`;

// 式の埋め込み
const price = 1200;
const tax = 0.1;
const total = `合計: ${price * (1 + tax)}円`;

非同期処理の実践マスター

Promiseから async/await への進化

Promiseは非同期処理の結果を表すオブジェクトですが、async/awaitにより同期的な書き方が可能になりました。

Promise連鎖の問題

// Promise連鎖（読みにくい）
function fetchUserData(userId) {
    return fetch(`/api/users/${userId}`)
        .then(response => response.json())
        .then(user => {
            return fetch(`/api/posts/${user.id}`);
        })
        .then(response => response.json())
        .then(posts => {
            return fetch(`/api/comments/${posts[0].id}`);
        })
        .then(response => response.json())
        .catch(error => {
            console.error('エラー:', error);
        });
}

async/await（読みやすい）

// async/await（読みやすい）
async function fetchUserData(userId) {
    try {
        const userResponse = await fetch(`/api/users/${userId}`);
        const user = await userResponse.json();
        
        const postsResponse = await fetch(`/api/posts/${user.id}`);
        const posts = await postsResponse.json();
        
        const commentsResponse = await fetch(`/api/comments/${posts[0].id}`);
        const comments = await commentsResponse.json();
        
        return {user, posts, comments};
    } catch (error) {
        console.error('エラー:', error);
        throw error;
    }
}

並列処理と逐次処理 

// 並列処理（Promise.all）- 高速
async function fetchAllDataInParallel() {
    try {
        const [users, posts, comments] = await Promise.all([
            fetch('/api/users').then(r => r.json()),
            fetch('/api/posts').then(r => r.json()),
            fetch('/api/comments').then(r => r.json())
        ]);
        
        console.log('すべてのデータを並列取得完了');
        return {users, posts, comments};
    } catch (error) {
        console.error('並列処理エラー:', error);
    }
}

// 逐次処理 - 順序が重要な場合
async function fetchDataSequentially() {
    try {
        const users = await fetch('/api/users').then(r => r.json());
        console.log('ユーザー取得完了');
        
        const posts = await fetch(`/api/posts?userId=${users[0].id}`).then(r => r.json());
        console.log('投稿取得完了');
        
        const comments = await fetch(`/api/comments?postId=${posts[0].id}`).then(r => r.json());
        console.log('コメント取得完了');
        
        return {users, posts, comments};
    } catch (error) {
        console.error('逐次処理エラー:', error);
    }
}
