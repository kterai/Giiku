メディアクエリの基本構文

/* モバイルファースト設計 */
/* ベースはモバイル向けスタイル */
.container {
    width: 100%;
    padding: 10px;
}

/* タブレット向け (768px以上) */
@media (min-width: 768px) {
    .container {
        width: 750px;
        margin: 0 auto;
        padding: 20px;
    }
}

/* デスクトップ向け (992px以上) */
@media (min-width: 992px) {
    .container {
        width: 970px;
        padding: 30px;
    }
}

/* 大型画面向け (1200px以上) */
@media (min-width: 1200px) {
    .container {
        width: 1170px;
    }
}
