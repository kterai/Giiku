HTML構造例

<button onclick="toggleAnswer('q1')" class="mt-2 bg-blue-500 text-white px-4 py-2 rounded hover:bg-blue-600">
    回答を表示
</button>
<div id="q1-answer" class="hidden mt-2 p-4 bg-green-50 border-l-4 border-green-500">
    <p class="font-semibold text-green-800">正解: (選択肢)</p>
    <p class="text-green-700">解説...</p>
</div>

JavaScript機能

function toggleAnswer(questionId) {
    const element = document.getElementById(questionId + '-answer');
    const button = event.target;
    
    if (element.classList.contains('hidden')) {
        element.classList.remove('hidden');
        button.textContent = '回答を非表示';
        button.classList.add('bg-gray-500');
        button.classList.remove('bg-blue-500');
    } else {
        element.classList.add('hidden');
        button.textContent = '回答を表示';
        button.classList.add('bg-blue-500');
        button.classList.remove('bg-gray-500');
    }
}
